Examples

Usage example for estimating the self-similarity of a time series generated with the Hosking method using the Rescaled Range Statistics. 
#include "proc.h"
#include "gen.h"
#include "est.h"
#include "dist.h"
#include "stat.h"
#include "rng.h"
#include "io.h"
#include "util.h"
#include "batt.h"
#include "sim.h"

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define N   pow(2,15)
#define H   (double) 0.25

int main (int argc, char *argv[]) {

    proc_Process *pr = gen_Hosking (N, H, TestH_fBm);

    proc_ScalesConfig *conf = proc_CreateScalesConfig (TestH_POW, 7, 11, 2);

    proc_CreateScales (pr, conf);

    est_RescaledRangeStatistics (pr);

    proc_DeleteProcess (pr);

    return 1;
}
